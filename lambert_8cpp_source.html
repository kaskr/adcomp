4 // Double version of Lambert W function
5 double LambertW(double x) {
6  double logx = log(x);
7  double y = (logx > 0 ? logx : 0);
8  int niter = 100, i=0;
9  for (; i < niter; i++) {
10  if ( fabs( logx - log(y) - y) < 1e-9) break;
11  y -= (y - exp(logx - y)) / (1 + y);
12  }
13  if (i == niter) Rf_warning("W: failed convergence");
14  return y;
15 }
