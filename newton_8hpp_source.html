109 struct HessianSolveVector : TMBad::global::DynamicOperator< -1, -1 > {
110  static const bool have_input_size_output_size = true;
111  static const bool add_forward_replay_copy = true;
112  std::shared_ptr<Hessian_Type> hessian;
113  size_t nnz, x_rows, x_cols; // Dim(x)
114  HessianSolveVector(std::shared_ptr<Hessian_Type> hessian, size_t x_cols = 1) :
115  hessian ( hessian ),
116  nnz ( hessian->Range() ),
117  x_rows ( hessian->n ),
118  x_cols ( x_cols ) {}
119  TMBad::Index input_size() const {
120  return nnz + x_rows * x_cols;
121  }
122  TMBad::Index output_size() const {
123  return x_rows * x_cols;
124  }
125  vector<TMBad::Scalar> solve(const vector<TMBad::Scalar> &h,
126  const vector<TMBad::Scalar> &x) {
127  typename Hessian_Type::template MatrixResult<TMBad::Scalar>::type
128  H = hessian -> as_matrix(h);
129  hessian -> llt_factorize(H); // Assuming analyzePattern(H) has been called once
130  matrix<TMBad::Scalar> xm = x.matrix();
131  xm.resize(x_rows, x_cols);
132  vector<TMBad::Scalar> y = hessian -> llt_solve(H, xm).vec();
133  return y;
134  }
135  vector<TMBad::Replay> solve(const vector<TMBad::Replay> &h,
136  const vector<TMBad::Replay> &x) {
137  std::vector<TMBad::ad_plain> hx;
138  hx.insert(hx.end(), h.data(), h.data() + h.size());
139  hx.insert(hx.end(), x.data(), x.data() + x.size());
140  TMBad::global::Complete<HessianSolveVector> Op(*this);
141  std::vector<TMBad::ad_plain> ans = Op(hx);
142  std::vector<TMBad::ad_aug> ans2(ans.begin(), ans.end());
143  return ans2;
144  }
145  void forward(TMBad::ForwardArgs<TMBad::Scalar> &args) {
146  size_t n = output_size();
147  vector<TMBad::Scalar>
148  h = args.x_segment(0, nnz),
149  x = args.x_segment(nnz, n);
150  args.y_segment(0, n) = solve(h, x);
151  }
152  template <class T>
153  void reverse(TMBad::ReverseArgs<T> &args) {
154  size_t n = output_size();
155  vector<T>
156  h = args. x_segment(0, nnz),
157  y = args. y_segment(0, n),
158  dy = args.dy_segment(0, n);
159  vector<T> y2 = solve(h, dy);
160  for (size_t j=0; j < x_cols; j++) {
161  vector<T> y_j = y .segment(j * x_rows, x_rows);
162  vector<T> y2_j = y2.segment(j * x_rows, x_rows);
163  vector<T> y2y_j = hessian -> crossprod(y2_j, y_j);
164  args.dx_segment(0, nnz) -= y2y_j;
165  args.dx_segment(nnz + j * x_rows, x_rows) += y2_j;
166  }
167  }

772  // Outer parameters
773  std::vector<TMBad::ad_aug> par_outer;
779  NewtonOperator(Functor &F, vector<TMBad::ad_aug> start, newton_config cfg)
780  : cfg(cfg)
781  {
782  // Create tape of the user functor and optimize the tape
783  function = TMBad::ADFun<> ( FunctorExtend(F), start);
784  function.optimize();
785  // The tape may contain expressions that do not depend on the
786  // inner parameters (RefOp and expressions that only depend on
787  // these). Move such expressions to the parent context?
788  if (cfg.decompose) {
789  function.decompose_refs();
790  }
791  // The previous operation does not change the domain vector size.
792  size_t n_inner = function.Domain();
793  TMBAD_ASSERT(n_inner == (size_t) start.size());
794  // Turn remaining references to parent contexts into outer
795  // parameters. This operation increases function.Domain()
796  par_outer = function.resolve_refs();
797  // Mark inner parameter subset in full parameter vector
798  std::vector<bool> keep_inner(n_inner, true);
799  keep_inner.resize(function.Domain(), false);
800  // Gradient function
801  gradient = function.JacFun(keep_inner);
802  if (cfg.simplify) {
803  // Masks
804  std::vector<bool> active = gradient.activeDomain();
805  for (size_t i=0; i<n_inner; i++) active[i] = true; // just in case ...
806  size_t num_inactive = std::count(active.begin(), active.end(), false);
807  if (cfg.trace) {
808  std::cout << "Dead gradient args to 'simplify': ";
809  std::cout << num_inactive << "\n";
810  }
811  if (num_inactive > 0) {
812  function.DomainReduce(active);
813  gradient.DomainReduce(active);
814  std::vector<bool> active_outer(active.begin() + n_inner, active.end());
815  par_outer = TMBad::subset(par_outer, active_outer);
816  TMBAD_ASSERT(n_inner == (size_t) function.inner_inv_index.size());
817  function.optimize();
818  }
819  }
820  gradient.optimize();
821  // Hessian
822  hessian = std::make_shared<Hessian_Type>(function, gradient, n_inner);
823  hessian -> optimize();
824  }

840  // Append outer paramaters to solution

846  return 1. / u - 1.;

853  }
854  double decrease(double u) {

856  1. - increase(1. - u) :
857  (1. - cfg.u0) * cfg.power * u);

874  const char* newton_iterate(vector<Scalar> &x) {
875  int reject_counter = 0;
876  const char* msg = NULL;
877  Scalar f_x = function(x)[0];
878  if (x_start.size() == x.size()) {
879  Scalar f_x_start = function(x_start)[0];
880  if ( ! std::isfinite(f_x_start) &&
881  ! std::isfinite(f_x) ) {
882  return
883  convergence_fail("Invalid initial guess", x);
884  }
885  if (f_x_start < f_x || ! std::isfinite(f_x)) {
886  x = x_start;
887  f_x = f_x_start;
888  }
889  }
890  Scalar f_previous = f_x;
891  if (cfg.trace) std::cout << "f_start=" << f_x << "\n";
892  for (int i=0; i < cfg.maxit; i++) {
893  vector<Scalar> g = gradient(x);
894  Scalar mgc = g.abs().maxCoeff();
895  if ( ! std::isfinite(mgc) ||
896  mgc > cfg.mgcmax) {
897  return
898  convergence_fail("Inner gradient had non-finite components", x);
899  }
900  /* FIXME:
901  if (any(!is.finite(g)))
902  stop("Newton dropout because inner gradient had non-finite components.")
903  if (is.finite(mgcmax) && max(abs(g)) > mgcmax)
904  stop("Newton dropout because inner gradient too steep.")
905  if (max(abs(g)) < grad.tol)
906  return(par)
907  */
908  if (cfg.trace) std::cout << "mgc=" << mgc << " ";
909  if (mgc < cfg.grad_tol) {
910  x_start = x;
911  if (cfg.trace) std::cout << "\n";
912  return msg;
913  }
914  typename Hessian_Type::template MatrixResult<TMBad::Scalar>::type
915  H = (*hessian)(std::vector<Scalar>(x));
916  vector<Scalar> diag_cpy = H.diagonal().array();
917  // Quick ustep reduction based on Hessian diagonal
918  Scalar m = diag_cpy.minCoeff();
919  if (std::isfinite(m) && m < 0) {
920  Scalar ustep_max = invphi(-m);
921  cfg.ustep = std::min(cfg.ustep, ustep_max);
922  }
923  if (cfg.trace) std::cout << "ustep=" << cfg.ustep << " ";
924  while (true) { // FIXME: Infinite loop
925  // H := H + phi * I
926  H.diagonal().array() = diag_cpy;
927  H.diagonal().array() += phi( cfg.ustep );
928  // Try to factorize
929  hessian -> llt_factorize(H);
930  if (hessian -> llt_info() == 0) break;
931  // H not PD ==> Decrease phi
932  cfg.ustep = decrease(cfg.ustep);
933  }
934  // We now have a PD hessian
935  // Let's take a newton step and see if it improves...
936  vector<Scalar> x_new =
937  x - hessian -> llt_solve(H, g).array();
938  Scalar f = function(x_new)[0];
939  // Accept/Reject rules
940  bool accept = std::isfinite(f);
941  if ( ! accept ) {
942  reject_counter++;
943  } else {
944  // Accept if there's a value improvement:
945  accept =
946  (f < f_previous);
947  // OR...
948  if (! accept) {
949  // Accept if relative mgc reduction is substantial without increasing value 'too much'
950  accept =
951  (f - f_previous <= cfg.signif_abs_reduction ) &&
952  vector<Scalar>(gradient(x_new)).abs().maxCoeff() / mgc < 1. - cfg.signif_rel_reduction;
953  }
954  // Assess the quality of the update
955  if ( accept && (f_previous - f > cfg.signif_abs_reduction) ) {
956  reject_counter = 0;
957  } else {
958  reject_counter++;
959  }
960  }
961  // Take action depending on accept/reject
962  if ( accept ) { // Improvement
963  // Accept
964  cfg.ustep = increase(cfg.ustep);
965  f_previous = f;
966  x = x_new;
967  } else { // No improvement
968  // Reject
969  cfg.ustep = decrease(cfg.ustep);
970  }
971  // Handle long runs without improvement
972  if (reject_counter > cfg.max_reject) {
973  if (cfg.ok_exit_if_pdhess) {
974  H = (*hessian)(std::vector<Scalar>(x));
975  // Try to factorize
976  hessian -> llt_factorize(H);
977  bool PD = (hessian -> llt_info() == 0);
978  if (cfg.trace) std::cout << "Trying early exit - PD Hess? " << PD << "\n";
979  if (PD) return msg;
980  }
981  return
982  convergence_fail("Max number of rejections exceeded", x);
983  }
984  // Tracing info
985  if (cfg.trace) std::cout << "f=" << f << " ";
986  if (cfg.trace) std::cout << "reject=" << reject_counter << " ";
987  if (cfg.trace) std::cout << "\n";
988  }
989  return
990  convergence_fail("Iteration limit exceeded", x);
991  }
