Calculate gradient of vector function with scalar values.

Parameters
FFunctor with scalar values
xVector evaluation point
Returns
Gradient vector
Note
The evaluation method of the functor must be templated

Example:

#include <TMB.hpp>
struct func {
template <class T>
T operator()(vector<T> x){ // Evaluate function
return exp( -x.sum() );
}
};
template<class Type>
Type objective_function<Type>::operator() () {
PARAMETER_VECTOR(theta);
func f;
// Calculate gradient
vector<Type> g = autodiff::gradient(f, theta);
REPORT(g);
// Exit
return 0;
}
