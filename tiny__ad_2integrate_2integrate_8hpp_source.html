105  int a_finite = (a_ != -INFINITY) && (a_ != INFINITY);
106  int b_finite = (b_ != -INFINITY) && (b_ != INFINITY);
107  if ( a_finite && b_finite) { inf = 0; a = a_; b = b_; }
108  else if ( a_finite && !b_finite) { inf = 1; bound = a_; }
