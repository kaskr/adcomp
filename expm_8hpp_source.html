5  /* Matrix class with methods required by Pade approximation */
6  template<class Type>
7  struct Block {
8  typedef Type ScalarType;
9  matrix<Type> A;
10  Block(){}
11  Block(matrix<Type> A_){A=A_;}
12  /* Matrix multiply */
13  Block<Type> operator*(Block<Type> other){
14  return Block( this->A * other.A );
15  }
16  /* Scale matrix */
17  Block<Type> scale(Type c){
18  return Block( c * this->A );
19  }
20  /* Add identity matrix */
21  Block<Type> addIdentity(){
22  int n=A.rows();
23  matrix<double> I(n,n);
24  I.setIdentity();
25  return Block( this->A + I );
26  }
27  /* Infinity norm */
28  Type norm(){
29  matrix<Type> Aabs(this->A.rows(),this->A.cols());
30  Aabs = A.array().abs();
31  vector<Type> rsAabs = Aabs.rowwise().sum();
32  return rsAabs.maxCoeff();
33  }
34  /* Matrix inverse */
35  Block<Type> inverse(){
36  return Block( this->A.inverse() );
37  }
38  /* Increment/decrement */
39  Block<Type>& operator+=(Block<Type> x){ this->A += x.A; return *this; }
40  Block<Type>& operator-=(Block<Type> x){ this->A -= x.A; return *this; }
41  /* Methods for operator square root */
42  typedef Eigen::SelfAdjointEigenSolver<
43  Eigen::Matrix<Type, Eigen::Dynamic, Eigen::Dynamic> > SAES_t;
44  /* Solve *special case* of Sylvester equation: XA+AX=Y */
45  Block<Type> sylvester(Block<Type> Y) {
46  SAES_t saes(A);
47  matrix<Type> V = saes.eigenvectors();
48  vector<Type> D = saes.eigenvalues();
49  // Transform
50  matrix<Type> Y_ = V.transpose() * Y.A * V;
51  // Solve
52  for (int i=0; i<Y_.rows(); i++)
53  for (int j=0; j<Y_.cols(); j++)
54  Y_(i, j) /= (D(i) + D(j));
55  // Transform back
56  matrix<Type> X = V * Y_ * V.transpose();
57  return Block(X);
58  }
59  /* Operator square root (PD case only) */
60  Block<Type> sqrtm() {
61  SAES_t saes(A);
62  matrix<Type> X = saes.operatorSqrt();
63  return Block(X);
64  }
65  };

67  /*
68  Representation of matrix of the form
69 
70  T(A,B) :=
71 
72  [ A 0 ]
73  [ B A ]
74 
75  with methods required by Pade approximation
76  */
77  template<class BlockType>
78  struct Triangle {
79  BlockType A, B;
80  typedef typename BlockType::ScalarType Type;
81  Triangle(){}
82  Triangle(BlockType A_, BlockType B_){A = A_; B = B_;}
83  /* Matrix multiply. Cost = 3 multiplies */
84  Triangle<BlockType> operator*(Triangle<BlockType> other){
85  BlockType A, B;
86  A = (this->A) * (other.A);
87  B = (this->A) * (other.B);
88  B += (this->B) * (other.A);
89  return Triangle(A, B);
90  }
91  /* Scale matrix */
92  Triangle<BlockType> scale(Type c){
93  return Triangle(this->A.scale(c), this->B.scale(c));
94  }
95  /* Add identity matrix */
96  Triangle<BlockType> addIdentity(){
97  return Triangle(this->A.addIdentity(), this->B);
98  }
99  /* Infinity norm (Strictly, the norm is larger. But we
100  wan't the number of pade iterations for the derivative
101  to be the same as for the function value) */
102  Type norm(){
103  return this->A.norm();
104  }
105  /* Matrix inverse. Cost = 1 inverse + 2 multiplies */
106  Triangle<BlockType> inverse(){
107  BlockType A = this->A.inverse();
108  BlockType B = (A * (this->B * A)).scale(-1);
109  return Triangle(A, B);
110  }
111  /* Increment/decrement */
112  Triangle<BlockType>& operator+=(Triangle<BlockType> x){
113  this->A += x.A;
114  this->B += x.B;
115  return *this;
116  }
117  Triangle<BlockType>& operator-=(Triangle<BlockType> x){
118  this->A -= x.A;
119  this->B -= x.B;
120  return *this;
121  }
122  /* Methods for operator square root */
123  Triangle<BlockType> sylvester(Triangle<BlockType> Y) {
124  Triangle<BlockType> X;
125  X.A = (*this).A.sylvester(Y.A);
126  Y.B -= (*this).B * X.A;
127  Y.B -= X.A * (*this).B;
128  X.B = (*this).A.sylvester(Y.B);
129  return X;
130  }
131  Triangle<BlockType> sqrtm() {
132  BlockType A = (*this).A.sqrtm();
133  BlockType B = A.sylvester((*this).B);
134  return Triangle(A, B);
135  }
136  };

158  struct nestedTriangle : Triangle<nestedTriangle<n-1> >{
159  typedef double ScalarType;
160  typedef nestedTriangle<n-1> BlockType;
161  typedef Triangle<nestedTriangle<n-1> > Base;
162  nestedTriangle(){}

164  nestedTriangle(vector<matrix<double> > args) : Base() {
165  int nargs=args.size();
166  vector<matrix<double> > args1 = args.head(nargs-1);
167  matrix<double> zero = args[0]*0.0;
168  vector<matrix<double> > args2(nargs-1);
169  for(int i=0;i<nargs-1;i++)args2[i] = zero;
170  args2[0] = args[nargs-1];
171  Base::A = nestedTriangle<n-1>(args1);
172  Base::B = nestedTriangle<n-1>(args2);
173  }
174  /* For easy recursive extraction */
175  matrix<double> bottomLeftCorner(){
176  return this->B.bottomLeftCorner();
177  }

191  /*
192  Pade approximation of matrix exponential.
193  Can be applied to any of the previously implemented matrix
194  classes.
195  */
196  template<class matrix_pade>
197  matrix_pade expm(matrix_pade A){
198  double log2NormInf = log( A.norm() );
199  log2NormInf /= log(double(2.0));
200  double e = std::floor(log2NormInf) + 1.0;
201  double s = e + 1.0;
202  s = (s < 0 ? 0 : s);
203  matrix_pade AA = A.scale( 1.0 / pow( 2.0, s ) );
204  matrix_pade X = AA;
205  double c = 0.5;
206  matrix_pade E = AA.scale( c).addIdentity();
207  matrix_pade D = AA.scale(-c).addIdentity();
208  int q = 8; /* R uses 8 - ADMB 6 */
209  int p = 1;
210  for(int k = 2; k<=q; ++k){
211  c *= double(q-k+1) / double(k*(2*q-k+1));
212  X = AA * X;
213  matrix_pade cX = X.scale(c);
214  E += cX;
215  if (p == 1) { D += cX; } else { D -= cX; }
216  p = (p == 1) ? 0 : 1;
217  }
218  matrix_pade invD = D.inverse();
219  E = invD * E;
220  for(int k = 1; k<=s; k++){
221  E = E * E;
222  }
223  return E;
224  }

226  /* Matrix exponential and derivatives up to (in principle) any order.
227  Orders from 0 to 3 are compiled and the appropriate order is dispatched
228  at run-time.
229  */
230  matrix<double> expm(vector<matrix<double> > args)CSKIP({
231  int nargs = args.size();
232  matrix<double> ans;
233  if (nargs==1) ans=expm(nestedTriangle<0>(args)).bottomLeftCorner();
234  else if (nargs==2) ans=expm(nestedTriangle<1>(args)).bottomLeftCorner();
235  else if (nargs==3) ans=expm(nestedTriangle<2>(args)).bottomLeftCorner();
236  else if (nargs==4) ans=expm(nestedTriangle<3>(args)).bottomLeftCorner();
237  else Rf_error("expm: order not implemented.");
238  return ans;

241  /* Helper to convert list of matrices to CppAD::vector
242  of the format: 'c(length(args), unlist(args))'
243  */
244  template<class Type>
245  CppAD::vector<Type> args2vector(vector<matrix<Type> > args, int skip=-1){
246  int nargs = args.size();
247  int nkeep = nargs - (skip >= 0);
248  int blocksize = args[0].size();
249  int n = 1 + nkeep * blocksize;
250  CppAD::vector<Type> ans(n);
251  ans[0] = nkeep;
252  int k=1;
253  for(int i=0; i<nargs; i++){
254  if(i != skip){
255  for(int j=0; j < blocksize; j++){
256  ans[k] = args[i](j);
257  k++;
258  }
259  }
260  }
261  return ans;
262  }
