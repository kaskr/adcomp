250  vector<int> tuple(int i){
251  vector<int> revtup(dim.size());
252  vector<int> revmult=mult.reverse();
253  for(int k=0;k<dim.size();k++){
254  revtup[k]=i/revmult[k];
255  i=i-revtup[k]*revmult[k];
