672  /* ================== For parallel Hessian computation
673  Need three different parallel evaluation modes:
674  (1) *Parallel mode* where a parallel region is evaluated iff
675  current_parallel_region == selected_parallel_region
676  (2) *Serial mode* where all parallel region tests are evaluated
677  to TRUE so that "PARALLEL_REGION" tests are effectively removed.
678  A negative value of "current_parallel_region" or "selected_parallel_region"
679  is used to select this mode (the default).
680  (3) *Count region mode* where statements inside "PARALLEL_REGION{...}"
681  are *ignored* and "current_parallel_region" is increased by one each
682  time a parallel region is visited.
683  NOTE: The macro "PARALLEL_REGION" is supposed to be defined as
684  #define PARALLEL_REGION if(this->parallel_region())

689  int selected_parallel_region; /* Consider _this_ code-fragment */
690  int max_parallel_regions; /* Max number of parallel region identifiers,
691  e.g. max_parallel_regions=omp_get_max_threads();
692  probably best in most cases. */
693  bool parallel_region(){ /* Is this the selected parallel region ? */
694  bool ans;
695  if(current_parallel_region<0 || selected_parallel_region<0)return true; /* Serial mode */
696  ans = (selected_parallel_region==current_parallel_region) && (!parallel_ignore_statements);
697  current_parallel_region++;
698  if(max_parallel_regions>0)current_parallel_region=current_parallel_region % max_parallel_regions;
699  return ans;

1509 sphess MakeADHessObject2_(SEXP data, SEXP parameters, SEXP report, SEXP skip, int parallel_region=-1)
1510 {
1511  /* Some type checking */
1512  if(!Rf_isNewList(data))Rf_error("'data' must be a list");
1513  if(!Rf_isNewList(parameters))Rf_error("'parameters' must be a list");
1514  if(!Rf_isEnvironment(report))Rf_error("'report' must be an environment");
1515 
1516  /* Prepare stuff */
1517  objective_function< AD<AD<AD<double> > > > F(data,parameters,report);
1518  F.set_parallel_region(parallel_region);
1519  int n = F.theta.size();
1520  vector<bool> keepcol(n); // Scatter for fast lookup
1521  for(int i=0; i<n; i++){
1522  keepcol[i]=true;
1523  }
1524  for(int i=0; i<LENGTH(skip); i++){
1525  keepcol[INTEGER(skip)[i]-1]=false; // skip is R-index !
1526  }
1527 #define KEEP_COL(col) (keepcol[col])
1528 #define KEEP_ROW(row,col) ( KEEP_COL(row) & (row>=col) )
1529 
1530  /* Tape 1: Function R^n -> R */
1531  Independent(F.theta);
1532  vector< AD<AD<AD<double> > > > y(1);
1533  y[0] = F.evalUserTemplate();
1534  ADFun<AD<AD<double> > > tape1(F.theta, y);
1535 
1536  /* Tape 2: Gradient R^n -> R^n (and optimize) */
1537  vector<AD<AD<double> > > xx(n);
1538  for(int i=0; i<n; i++) xx[i] = CppAD::Value(F.theta[i]);
1539  vector<AD<AD<double> > > yy(n);
1540  Independent(xx);
1541  yy = tape1.Jacobian(xx);
1542  ADFun<AD<double > > tape2(xx,yy);
1543  if (config.optimize.instantly) tape2.optimize();
1544 
1545  /* Tape 3: Hessian R^n -> R^m (optimize later) */
1546  tape2.my_init(keepcol);
1547  int colisize;
1548  int m=0; // Count number of non-zeros (m)
1549  for(int i=0; i<int(tape2.colpattern.size()); i++){
1550  colisize = tape2.colpattern[i].size();
1551  if(KEEP_COL(i)){
1552  for(int j=0; j<colisize; j++){
1553  m += KEEP_ROW( tape2.colpattern[i][j] , i);
1554  }
1555  }
1556  }
1557  // Allocate index vectors of non-zero pairs
1558  vector<int> rowindex(m);
1559  vector<int> colindex(m);
1560  // Prepare reverse sweep for Hessian columns
1561  vector<AD<double> > u(n);
1562  vector<AD<double> > v(n);
1563  for(int i = 0; i < n; i++) v[i] = 0.0;
1564  vector<AD<double> > xxx(n);
1565  for(int i=0; i<n; i++) xxx[i]=CppAD::Value(CppAD::Value(F.theta[i]));
1566  vector<AD<double> > yyy(m);
1567  CppAD::vector<int>* icol;
1568  // Do sweeps and fill in non-zero index pairs
1569  Independent(xxx);
1570  tape2.Forward(0, xxx);
1571  int k=0;
1572  for(int i = 0; i < n; i++){
1573  if (KEEP_COL(i)) {
1574  tape2.myReverse(1, v, i /*range comp*/, u /*domain*/);
1575  icol = &tape2.colpattern[i];
1576  for(int j=0; j<int(icol->size()); j++){
1577  if(KEEP_ROW( icol->operator[](j), i )){
1578  rowindex[k] = icol->operator[](j);
1579  colindex[k] = i;
1580  yyy[k] = u[icol->operator[](j)];
1581  k++;
1582  }
1583  }
1584  }
1585  }
1586  ADFun< double >* ptape3 = new ADFun< double >;
1587  ptape3->Dependent(xxx,yyy);
1588  sphess ans(ptape3, rowindex, colindex);
1589  return ans;
