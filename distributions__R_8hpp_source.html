184 {
185  Type logres = lgamma(size + 1) - lgamma(k + 1) - lgamma(size - k + 1);
186  // Add 'k * log(prob)' only if k > 0

188  // Add '(size - k) * log(1 - prob)' only if size > k

550 template<class Type>
551 Type dtweedie(Type y, Type mu, Type phi, Type p, int give_log = 0) {
552  Type p1 = p - 1.0, p2 = 2.0 - p;
553  Type ans = -pow(mu, p2) / (phi * p2); // log(prob(y=0))
554  if (y > 0) {
555  CppAD::vector<Type> tx(4);
556  tx[0] = y;
557  tx[1] = phi;
558  tx[2] = p;
559  tx[3] = 0;
560  ans += atomic::tweedie_logW(tx)[0];
561  ans += -y / (phi * p1 * pow(mu, p1)) - log(y);
562  }
563  return ( give_log ? ans : exp(ans) );
564 }

822 Type rtweedie(Type mu, Type phi, Type p) {

824  Type lambda = pow(mu, 2. - p) / (phi * (2. - p));
